#include<stdio.h>
#include<conio.h>
void main()
{
 int a, i, j;
 printf("Enter a value:");
 scanf("%d", &a);
 for(i=1; i<=a; i++)
 {
  for(j=a; j>(a-i); j--)
  {
   printf("%d", j);
  }
  for(j=1; j<=(a-i); j++)
  {
   printf("%d", a-(i+1));
  }
  printf("/n");
 } 
 return 0;
}
